    다음으로 CSV 샘플 데이터를 Delta 테이블로 저장합니다.

    ```python
    from pyspark.sql import functions as F

    # 샘플 시청 이력 데이터 로드
    df_raw = spark.read.option("header", True).csv(
        "/Volumes/smarttv_workshop/assets/watch_history_sample.csv"
    )

    # 타입 변환 및 기본 전처리
    df_clean = (
        df_raw
        .withColumn("watch_duration_min", F.col("watch_duration_min").cast("int"))
        .withColumn("rating",             F.col("rating").cast("double"))
        .withColumn("watch_ts",           F.to_timestamp("watch_ts"))
        .withColumn("date",               F.to_date("watch_ts"))
    )

    # Delta 테이블로 저장
    df_clean.write.format("delta").mode("overwrite").saveAsTable("watch_history")
    print(f"✓ 저장 완료: {df_clean.count():,}개 행")
    ```

  <Step title="피처 엔지니어링">
    시청 이력을 바탕으로 사용자 취향 피처를 생성합니다.

    ```python
    # 사용자별 선호 장르 및 시청 패턴 계산
    df_user_features = (
        spark.table("watch_history")
        .groupBy("user_id")
        .agg(
            F.count("content_id").alias("total_views"),
            F.avg("watch_duration_min").alias("avg_watch_min"),
            F.avg("rating").alias("avg_rating"),
            F.collect_set("genre").alias("preferred_genres"),
            F.max("watch_ts").alias("last_watched_ts"),
        )
        .withColumn(
            "engagement_score",
            F.round(F.col("avg_watch_min") * F.col("avg_rating") / 100, 2)
        )
    )

    df_user_features.write.format("delta").mode("overwrite").saveAsTable("user_features")
    display(df_user_features.limit(5))
    ```

    <Tip>
      `display()` 함수는 Databricks 노트북 전용입니다. 데이터프레임을 인터랙티브 테이블로 시각화해 주므로, `.show()` 대신 사용하세요.
    </Tip>
  </Step>

    ```python
    def build_recommendation_prompt(user_id: str, top_n: int = 5) -> str:
        """사용자 피처를 기반으로 추천 프롬프트를 생성합니다."""

        # 사용자 피처 조회
        user_feat = (
            spark.table("user_features")
            .filter(F.col("user_id") == user_id)
            .collect()
        )
        if not user_feat:
            raise ValueError(f"user_id '{user_id}'를 찾을 수 없습니다.")

        row = user_feat[0]
        genres = ", ".join(row["preferred_genres"][:5])

        # 최근 시청 콘텐츠 5개 조회
        recent_titles = (
            spark.table("watch_history")
            .filter(F.col("user_id") == user_id)
            .orderBy(F.col("watch_ts").desc())
            .limit(5)
            .select("title")
            .toPandas()["title"]
            .tolist()
        )

        prompt = f"""당신은 LG SmartTV 콘텐츠 추천 AI입니다.

사용자 정보:
- 선호 장르: {genres}
- 평균 시청 시간: {row['avg_watch_min']:.0f}분
- 평균 평점: {row['avg_rating']:.1f}/5.0
- 최근 시청 작품: {', '.join(recent_titles)}

위 정보를 바탕으로 이 사용자에게 적합한 콘텐츠 {top_n}편을 추천하고,
각각 한 문장으로 추천 이유를 설명해 주세요.
JSON 형식으로만 응답하세요:
[{{"rank": 1, "title": "...", "genre": "...", "reason": "..."}}]"""

        return prompt


    # 테스트 실행
    prompt = build_recommendation_prompt("user_0042", top_n=5)
    print(prompt)
    ```
  </Step>

  <Step title="추천 결과 생성 및 파싱">
    LLM을 호출하고 결과를 파싱합니다.

    ```python
    import json
    import re

    def get_recommendations(user_id: str, top_n: int = 5) -> list[dict]:
        """LLM을 호출해 추천 결과를 반환합니다."""
        prompt = build_recommendation_prompt(user_id, top_n)

        response = client.chat.completions.create(
            model="databricks-dbrx-instruct",
            messages=[
                {
                    "role": "system",
                    "content": "당신은 SmartTV 콘텐츠 추천 전문가입니다. JSON으로만 응답하세요."
                },
                {"role": "user", "content": prompt}
            ],
            max_tokens=1024,
            temperature=0.2,
        )

        raw = response.choices[0].message.content.strip()

        # JSON 추출 (마크다운 코드 블록 처리)
        json_match = re.search(r'\[.*\]', raw, re.DOTALL)
        if not json_match:
            raise ValueError(f"JSON 파싱 실패: {raw[:200]}")

        recommendations = json.loads(json_match.group())
        return recommendations


    # 테스트
    recs = get_recommendations("user_0042")
    for r in recs:
        print(f"{r['rank']}. {r['title']} ({r['genre']})\n   → {r['reason']}\n")
    ```

    <Tip>
      `temperature=0.2`로 낮게 설정하면 추천 결과가 일관성 있게 나옵니다. 창의적인 추천을 원한다면 `0.5~0.7` 범위로 조정해 보세요.
    </Tip>
  </Step>

  <Step title="MLflow로 실험 추적">
    여러 프롬프트 전략의 성능을 MLflow로 비교합니다.

    ```python
    import mlflow
    import time

    mlflow.set_experiment("/Users/<your-email>/smarttv-recommender")

    strategies = [
        {"name": "genre_based",   "top_n": 5, "temperature": 0.2},
        {"name": "balanced",      "top_n": 5, "temperature": 0.5},
        {"name": "exploratory",   "top_n": 5, "temperature": 0.8},
    ]

    for strategy in strategies:
        with mlflow.start_run(run_name=strategy["name"]):
            mlflow.log_params(strategy)

            start = time.time()
            recs = get_recommendations("user_0042", top_n=strategy["top_n"])
            latency_ms = (time.time() - start) * 1000

            mlflow.log_metric("latency_ms",    latency_ms)
            mlflow.log_metric("num_results",   len(recs))
            mlflow.log_text(json.dumps(recs, ensure_ascii=False, indent=2),
                            "recommendations.json")

            print(f"[{strategy['name']}] 지연시간: {latency_ms:.0f}ms, 결과: {len(recs)}개")
    ```
  </Step>
</Steps>

## 모듈 3: Model Serving 엔드포인트 배포

<Steps>
  <Step title="PyFunc 모델 클래스 정의">
    `03_model_serving.ipynb` 노트북을 열어 MLflow PyFunc 래퍼를 작성합니다.

    ```python
    import mlflow.pyfunc

    class SmartTVRecommender(mlflow.pyfunc.PythonModel):
        """Databricks Model Serving용 SmartTV 추천 모델."""

        def load_context(self, context):
            import os
            from openai import OpenAI

            self.client = OpenAI(
                api_key=os.environ["DATABRICKS_TOKEN"],
                base_url=f"{os.environ['DATABRICKS_HOST']}/serving-endpoints"
            )
            self.model_name = "databricks-dbrx-instruct"

        def predict(self, context, model_input):
            """
            model_input: pandas DataFrame
              - user_id (str): 추천을 요청할 사용자 ID
              - top_n   (int, optional): 추천 개수, 기본 5
            """
            import json, re, pandas as pd

            results = []
            for _, row in model_input.iterrows():
                user_id = row["user_id"]
                top_n   = int(row.get("top_n", 5))

                try:
                    recs = get_recommendations(user_id, top_n)
                    results.append({
                        "user_id":         user_id,
                        "recommendations": recs,
                        "status":          "ok"
                    })
                except Exception as e:
                    results.append({
                        "user_id":         user_id,
                        "recommendations": [],
                        "status":          f"error: {e}"
                    })

            return pd.DataFrame(results)
    ```
  </Step>

    Genie 플랫폼의 콘텐츠 요청 이벤트에 Databricks 엔드포인트 호출을 연결합니다.

    ```python
    import requests, os

    DATABRICKS_HOST = os.getenv("DATABRICKS_HOST")
    DATABRICKS_TOKEN = os.getenv("DATABRICKS_TOKEN")
    ENDPOINT_NAME   = os.getenv("SERVING_ENDPOINT_NAME")

    def fetch_recommendations_from_databricks(user_id: str, top_n: int = 5) -> list:
        """Databricks Serving 엔드포인트에서 추천 결과를 가져옵니다."""
        url = f"{DATABRICKS_HOST}/serving-endpoints/{ENDPOINT_NAME}/invocations"
        resp = requests.post(
            url,
            headers={
                "Authorization": f"Bearer {DATABRICKS_TOKEN}",
                "Content-Type":  "application/json",
            },
            json={"dataframe_records": [{"user_id": user_id, "top_n": top_n}]},
            timeout=10,
        )
        resp.raise_for_status()
        predictions = resp.json()["predictions"]
        return predictions[0]["recommendations"] if predictions else []


    @app.on_content_request
    def handle_content_request(req: ContentRequest) -> ContentResponse:
        """SmartTV에서 콘텐츠 추천 요청이 오면 호출됩니다."""
        recs = fetch_recommendations_from_databricks(
            user_id=req.user_id,
            top_n=req.requested_count or 5,
        )

        return ContentResponse(
            request_id=req.request_id,
            items=[
                {"title": r["title"], "genre": r["genre"], "reason": r["reason"]}
                for r in recs
            ]
        )

    print("✓ 핸들러 등록 완료")
    ```

      에뮬레이터 UI(`http://localhost:8888`)를 브라우저에서 열면 SmartTV 홈 화면에 추천 콘텐츠가 표시되는 것을 시각적으로 확인할 수 있습니다.
    </Tip>
