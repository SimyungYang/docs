AI 에이전트는 대형 언어 모델(LLM)이 외부 도구(Tool)를 호출하고 그 결과를 바탕으로 다음 행동을 결정하는 자율적 시스템입니다. Databricks는 Unity Catalog 기반의 데이터 거버넌스, Vector Search, MLflow 모델 추적, Model Serving을 통합하여 기업 환경에서 안전하게 AI 에이전트를 구축하고 운영하는 플랫폼을 제공합니다.

## RAG 에이전트 구축

### 1단계: 문서 임베딩 및 Databricks Vector Search 인덱스 생성

```python
from databricks.vector_search.client import VectorSearchClient
from langchain_community.document_loaders import PyPDFDirectoryLoader
from langchain.text_splitter import RecursiveCharacterTextSplitter
from langchain_community.embeddings import DatabricksEmbeddings

# 문서 로드 및 청크 분할
loader  = PyPDFDirectoryLoader("/Volumes/docs/product_manuals/")
docs    = loader.load()

splitter = RecursiveCharacterTextSplitter(
    chunk_size=1000,
    chunk_overlap=200,
    separators=["\n\n", "\n", ".", " "],
)
chunks = splitter.split_documents(docs)

# Delta 테이블로 청크 저장
from pyspark.sql import Row
chunks_df = spark.createDataFrame([
    Row(
        chunk_id=i,
        content=c.page_content,
        source=c.metadata.get("source", ""),
        page=c.metadata.get("page", 0),
    )
    for i, c in enumerate(chunks)
])
chunks_df.write.format("delta").mode("overwrite") \
    .saveAsTable("rag.vector_store.product_docs_chunks")

# Databricks Vector Search 엔드포인트 생성
vsc = VectorSearchClient()

# 엔드포인트가 없으면 생성
try:
    vsc.get_endpoint("rag-vector-endpoint")
except:
    vsc.create_endpoint(
        name="rag-vector-endpoint",
        endpoint_type="STANDARD",
    )

# Delta Sync 인덱스 생성 (소스 Delta 테이블과 자동 동기화)
index = vsc.create_delta_sync_index(
    endpoint_name="rag-vector-endpoint",
    source_table_name="rag.vector_store.product_docs_chunks",
    index_name="rag.vector_store.product_docs_index",
    pipeline_type="TRIGGERED",
    primary_key="chunk_id",
    embedding_source_column="content",
    embedding_model_endpoint_name="databricks-gte-large-en",
)
index.wait_until_ready()
print("Vector Search 인덱스 준비 완료")
```

### 2단계: LangChain RAG 체인 구성

```python
from langchain_community.vectorstores import DatabricksVectorSearch
from langchain_community.chat_models import ChatDatabricks
from langchain_community.embeddings import DatabricksEmbeddings
from langchain.chains import RetrievalQA
from langchain.prompts import PromptTemplate
from databricks.vector_search.client import VectorSearchClient

# 벡터 스토어 연결
vsc = VectorSearchClient()
vs_index = vsc.get_index(
    endpoint_name="rag-vector-endpoint",
    index_name="rag.vector_store.product_docs_index",
)

vectorstore = DatabricksVectorSearch(
    vs_index,
    text_column="content",
    embedding=DatabricksEmbeddings(endpoint="databricks-gte-large-en"),
    columns=["chunk_id", "source", "page"],
)
retriever = vectorstore.as_retriever(
    search_kwargs={"k": 5, "score_threshold": 0.7}
)

# 프롬프트 템플릿
prompt_template = PromptTemplate(
    input_variables=["context", "question"],
    template="""당신은 제품 전문가 어시스턴트입니다. 아래 컨텍스트를 바탕으로 질문에 정확하게 답하십시오.
컨텍스트에 답이 없으면 "제공된 문서에서 해당 정보를 찾을 수 없습니다"라고 답하십시오.

컨텍스트:
{context}

질문: {question}

답변:""",
)

# LLM 설정 (Databricks Model Serving의 DBRX 또는 외부 모델)
llm = ChatDatabricks(
    endpoint="databricks-dbrx-instruct",
    max_tokens=1024,
    temperature=0.1,
)

# RAG 체인 구성
rag_chain = RetrievalQA.from_chain_type(
    llm=llm,
    chain_type="stuff",
    retriever=retriever,
    return_source_documents=True,
    chain_type_kwargs={"prompt": prompt_template},
)

# 테스트
result = rag_chain.invoke({"query": "배터리 교체 방법은 무엇인가요?"})
print(result["result"])
print("\n참조 문서:")
for doc in result["source_documents"]:
    print(f"  - {doc.metadata['source']} (p.{doc.metadata['page']})")
```

## Tool-use 에이전트 구축

SQL 실행, 데이터 조회, 외부 API 호출을 LLM이 자율적으로 선택하여 실행하는 에이전트를 구성합니다.

```python
from langchain.agents import AgentExecutor, create_openai_tools_agent
from langchain.tools import tool
from langchain_community.chat_models import ChatDatabricks
from langchain.prompts import ChatPromptTemplate, MessagesPlaceholder
from databricks.sdk import WorkspaceClient

w = WorkspaceClient()

# --- 도구(Tool) 정의 ---

@tool
def execute_sql_query(query: str) -> str:
    """Databricks SQL Warehouse에서 SQL을 실행하고 결과를 반환합니다.
    
    Args:
        query: 실행할 SQL 쿼리 문자열
    
    Returns:
        쿼리 결과 (CSV 형식 문자열)
    """
    import pandas as pd
    statement = w.statement_execution.execute_statement(
        warehouse_id="<sql-warehouse-id>",
        statement=query,
        wait_timeout="30s",
    )
    if statement.status.state.value != "SUCCEEDED":
        return f"쿼리 실패: {statement.status.error}"

    columns = [col.name for col in statement.manifest.schema.columns]
    rows    = [[str(v) for v in row.values] for row in statement.result.data_array or []]
    df = pd.DataFrame(rows, columns=columns)
    return df.head(20).to_csv(index=False)


@tool
def get_table_schema(table_name: str) -> str:
    """지정한 테이블의 스키마(컬럼명, 타입, 설명)를 반환합니다.
    
    Args:
        table_name: 카탈로그.스키마.테이블 형식의 테이블 이름
    """
    result = w.statement_execution.execute_statement(
        warehouse_id="<sql-warehouse-id>",
        statement=f"DESCRIBE TABLE EXTENDED {table_name}",
        wait_timeout="10s",
    )
    rows = [[str(v) for v in row.values] for row in result.result.data_array or []]
    return "\n".join([f"{r[0]}: {r[1]} — {r[2]}" for r in rows if r[0] and not r[0].startswith("#")])


@tool
def get_available_tables() -> str:
    """분석에 사용 가능한 테이블 목록을 반환합니다."""
    result = w.statement_execution.execute_statement(
        warehouse_id="<sql-warehouse-id>",
        statement="SHOW TABLES IN gold.dw",
        wait_timeout="10s",
    )
    rows = [[str(v) for v in row.values] for row in result.result.data_array or []]
    return "\n".join([f"gold.dw.{r[1]}" for r in rows])


# --- 에이전트 구성 ---

tools = [execute_sql_query, get_table_schema, get_available_tables]

llm = ChatDatabricks(
    endpoint="databricks-meta-llama-3-1-70b-instruct",
    temperature=0,
)

prompt = ChatPromptTemplate.from_messages([
    ("system", """당신은 Databricks 레이크하우스의 데이터 분석 전문가입니다.
사용자의 비즈니스 질문에 답하기 위해 사용 가능한 도구로 데이터를 조회하십시오.
SQL을 작성하기 전에 먼저 테이블 스키마를 확인하고, 정확한 컬럼명을 사용하십시오.
결과는 한국어로 요약하여 비즈니스 인사이트를 제공하십시오."""),
    MessagesPlaceholder(variable_name="chat_history", optional=True),
    ("human", "{input}"),
    MessagesPlaceholder(variable_name="agent_scratchpad"),
])

agent = create_openai_tools_agent(llm=llm, tools=tools, prompt=prompt)

agent_executor = AgentExecutor(
    agent=agent,
    tools=tools,
    verbose=True,
    max_iterations=10,
    handle_parsing_errors=True,
)

# 에이전트 실행
result = agent_executor.invoke({
    "input": "지난 분기 대비 이번 분기 매출이 가장 많이 성장한 상위 3개 지역은 어디인가요?"
})
print(result["output"])
```

## MLflow로 에이전트 추적 및 배포

```python
import mlflow
from mlflow.models import infer_signature

mlflow.set_experiment("/Experiments/rag-agent")
mlflow.langchain.autolog()

# 에이전트를 MLflow 모델로 패키징
class RAGAgentModel(mlflow.pyfunc.PythonModel):
    def load_context(self, context):
        """모델 서빙 시 한 번만 실행 — 무거운 초기화 수행"""
        from langchain_community.vectorstores import DatabricksVectorSearch
        from langchain_community.chat_models import ChatDatabricks
        from databricks.vector_search.client import VectorSearchClient
        # ... 초기화 로직 ...
        self.chain = build_rag_chain()  # 위에서 정의한 체인

    def predict(self, context, model_input, params=None):
        """추론 엔드포인트 호출 시 실행"""
        questions = model_input["question"].tolist()
        results = []
        for q in questions:
            resp = self.chain.invoke({"query": q})
            results.append({
                "answer":  resp["result"],
                "sources": [d.metadata["source"] for d in resp["source_documents"]],
            })
        import pandas as pd
        return pd.DataFrame(results)


# 모델 등록
with mlflow.start_run(run_name="rag_agent_v1"):
    import pandas as pd
    example_input = pd.DataFrame({"question": ["배터리 교체 방법은?"]})
    example_output = pd.DataFrame({
        "answer":  ["배터리 덮개를 제거한 후..."],
        "sources": [["product_manual.pdf"]],
    })

    mlflow.pyfunc.log_model(
        artifact_path="rag_agent",
        python_model=RAGAgentModel(),
        registered_model_name="product-rag-agent",
        input_example=example_input,
        signature=infer_signature(example_input, example_output),
        pip_requirements=[
            "langchain>=0.2.0",
            "langchain-community>=0.2.0",
            "databricks-vectorsearch>=0.40",
        ],
    )
```

## Model Serving으로 에이전트 배포

```bash
# 에이전트 서빙 엔드포인트 생성
databricks serving-endpoints create \
  --name "product-rag-agent-endpoint" \
  --config '{
    "served_entities": [{
      "name": "rag-agent-v1",
      "entity_name": "product-rag-agent",
      "entity_version": "1",
      "workload_size": "Small",
      "scale_to_zero_enabled": true,
      "environment_vars": {
        "DATABRICKS_HOST":  "{{secrets/rag/databricks_host}}",
        "DATABRICKS_TOKEN": "{{secrets/rag/databricks_token}}"
      }
    }]
  }'
```

```python
import requests
import json

# 배포된 에이전트 호출
response = requests.post(
    url="https://<workspace>.azuredatabricks.net"
        "/serving-endpoints/product-rag-agent-endpoint/invocations",
    headers={
        "Authorization": "Bearer <token>",
        "Content-Type":  "application/json",
    },
    json={
        "dataframe_records": [
            {"question": "이 제품의 보증 기간은 얼마인가요?"},
            {"question": "고객 지원 연락처를 알려주세요."},
        ]
    },
)

predictions = response.json()["predictions"]
for pred in predictions:
    print(f"답변: {pred['answer']}")
    print(f"출처: {', '.join(pred['sources'])}\n")
```

## 에이전트 평가 (MLflow Evaluate)

```python
import mlflow
import pandas as pd

# 평가 데이터셋 준비
eval_data = pd.DataFrame({
    "question": [
        "배터리 교체 방법은 무엇인가요?",
        "제품 보증 기간은 얼마나 되나요?",
        "A/S 센터 운영 시간은 언제인가요?",
    ],
    "ground_truth": [
        "배터리 덮개를 제거하고 구형 배터리를 분리한 다음...",
        "구매일로부터 1년간 제조사 보증이 제공됩니다.",
        "평일 오전 9시부터 오후 6시까지 운영합니다.",
    ],
})

with mlflow.start_run(run_name="rag_agent_evaluation"):
    results = mlflow.evaluate(
        model=f"models:/product-rag-agent/1",
        data=eval_data,
        targets="ground_truth",
        model_type="question-answering",
        evaluators="default",
        extra_metrics=[
            mlflow.metrics.genai.answer_correctness(model="endpoints:/databricks-dbrx-instruct"),
            mlflow.metrics.genai.faithfulness(model="endpoints:/databricks-dbrx-instruct"),
            mlflow.metrics.genai.relevance(model="endpoints:/databricks-dbrx-instruct"),
        ],
    )
    print(results.metrics)
```
